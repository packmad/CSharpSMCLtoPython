declare client Millionaires:
	tunnelof sint netWorth;
	
	function void ask() {
		netWorth.put(readInt());
	}
	
	function void main() {
		ask();
	}
	
declare server Max:
	groupof Millionaires mills;
	
	function void main() {
		sint max;
		client rich;
		for (client c in mills) {
			sint netWorth = c.netWorth.get();
			c.ask();
			if (netWorth > max) {
				max = netWorth;
				rich = c;
			}
		}

	}
